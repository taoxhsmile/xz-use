# useToggle

A hook that switch value between two states.

The API is different from ahooks.

# Examples

# Basic usage

# API

const [state, { toggle, setLeft, setRight }] = useToggle(
  defaultValue?: boolean,
);

const [state, { toggle, setLeft, setRight }] = useToggle(
  defaultValue: any = false,
  reverseValue?: any,
);
